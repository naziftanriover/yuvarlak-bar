# Legend of the West - acilis (iki ekran, ayni profil -> canli hesap calisir)
Start-Sleep -Seconds 8

$U = "https://naziftanriover.github.io/yuvarlak-bar"

$chrome = @(
  "$env:ProgramFiles\Google\Chrome\Application\chrome.exe",
  "${env:ProgramFiles(x86)}\Google\Chrome\Application\chrome.exe",
  "$env:LocalAppData\Google\Chrome\Application\chrome.exe"
) | Where-Object { Test-Path $_ } | Select-Object -First 1
if (-not $chrome) { $chrome = "chrome.exe" }

Add-Type -AssemblyName System.Windows.Forms
$pri = [System.Windows.Forms.Screen]::PrimaryScreen.Bounds
$sec = [System.Windows.Forms.Screen]::AllScreens | Where-Object { -not $_.Primary } | Select-Object -First 1

Get-Process chrome -ErrorAction SilentlyContinue | Stop-Process -Force -ErrorAction SilentlyContinue
Start-Sleep -Seconds 2

# 1) MUSTERI EKRANI - ilk pencere (--app), arka ekrani kaplar
if ($sec) {
  Start-Process $chrome -ArgumentList "--kiosk-printing","--no-first-run","--disable-session-crashed-bubble","--window-position=$($sec.Bounds.X),$($sec.Bounds.Y)","--window-size=$($sec.Bounds.Width),$($sec.Bounds.Height)","--app=$U/musteri.html"
  Start-Sleep -Seconds 5
}

# 2) KASA - ayni profilde normal pencere (canli baglanti icin ayni profil sart)
Start-Process $chrome -ArgumentList "--new-window","$U/giris.html"
Start-Sleep -Seconds 7

# 3) Kasa penceresini bul ve ana (on) ekrana tasi + buyut.
# Kasa = basliginda " - Google Chrome" gecen normal pencere. (Musteri --app penceresinde gecmez.)
Add-Type @"
using System;
using System.Text;
using System.Runtime.InteropServices;
public class Win {
  public delegate bool EnumProc(IntPtr hWnd, IntPtr lParam);
  [DllImport("user32.dll")] public static extern bool EnumWindows(EnumProc lpEnumFunc, IntPtr lParam);
  [DllImport("user32.dll")] public static extern int GetWindowText(IntPtr hWnd, StringBuilder lpString, int nMaxCount);
  [DllImport("user32.dll")] public static extern bool IsWindowVisible(IntPtr hWnd);
  [DllImport("user32.dll")] public static extern bool MoveWindow(IntPtr hWnd, int X, int Y, int nWidth, int nHeight, bool bRepaint);
  [DllImport("user32.dll")] public static extern bool ShowWindow(IntPtr hWnd, int nCmdShow);
  public static IntPtr FindByTitle(string sub) {
    IntPtr found = IntPtr.Zero;
    EnumWindows((h, l) => {
      if (IsWindowVisible(h)) {
        StringBuilder sb = new StringBuilder(512);
        GetWindowText(h, sb, 512);
        if (sb.ToString().Contains(sub)) { found = h; return false; }
      }
      return true;
    }, IntPtr.Zero);
    return found;
  }
}
"@
$h = [Win]::FindByTitle(" - Google Chrome")
if ($h -ne [IntPtr]::Zero) {
  [Win]::ShowWindow($h, 9) | Out-Null
  [Win]::MoveWindow($h, $pri.X, $pri.Y, $pri.Width, $pri.Height, $true) | Out-Null
  [Win]::ShowWindow($h, 3) | Out-Null
}
